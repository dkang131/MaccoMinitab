MACRO

REGR Y X ALPHA

MCONSTANT N I Z T P PV PVALUE ai bi PV_A PV_B PVALUE_A PVALUE_B Ytaksir SE_A SE_B THIT_A THIT_B ALPHA ALPHA1 ALPHA2 TTAB XBAR YBAR JUMLAH1 JUMLAH2 A B YP SSR SSE SST MSR MSE FHIT FTAB DFE
MCOLUMN Y X

LET N=COUNT(X)
LET P=1
LET ALPHA1=1-(ALPHA)

LET JUMLAH1=0
LET JUMLAH2=0
DO I=1:N
	LET JUMLAH1=JUMLAH1+X(I)
	LET JUMLAH2=JUMLAH2+Y(I)
ENDDO
LET XBAR=JUMLAH1/N
LET YBAR=JUMLAH2/N

LET Z=0
LET T=0
DO I=1:N
	LET Z=Z+((X(I)-XBAR)*(Y(I)-YBAR))
	LET T=T+((X(I)-XBAR)^2)
ENDDO

LET B=Z/T
LET A=YBAR-(B*XBAR)

note nilai koefisien regresi
PRINT A B

# Y = A+BX

#ANOVA
LET SSR=0
LET YP=0
DO I=1:N
	LET YP=YP+(A+(B*X(I)))
	LET SSR=SSR+(((A+(B*X(I)))-YBAR)^2)
ENDDO

LET SSE=0
LET SST=0
DO I=1:N
	LET SSE=SSE+((Y(I)-(A+(B*X(I))))^2)
	LET SST=SST+((Y(I)-YBAR)^2)
ENDDO

LET DFE=N-P-1

LET MSR=SSR/P
LET MSE=SSE/(N-P-1)

PRINT SSR SSE SST MSR MSE

LET FHIT=MSR/MSE

NOTE ________________________________________________________________________
#UJI SERENTAK
NOTE UJI SERENTAK
INVCDF ALPHA1 FTAB;
	F P DFE.
PRINT FHIT FTAB

IF ABS(FHIT) > FTAB
NOTE 'TOLAK H0, MAKA KOEFISIEN PARAMETER MODEL REGRESI SIGNIFIKAN'
ELSE
NOTE 'GAGAL TOLAK H0, MAKA KOEFISIEN PARAMETER MODEL REGRESI SIGNIFIKAN'
ENDIF

#pvalue
let Fhit=abs(Fhit)
CDF Fhit pv;
	F P DFE.
let pvalue=1-PV
print pvalue alpha
if (pvalue < alpha)
NOTE 'TOLAK H0, MAKA KOEFISIEN PARAMETER MODEL REGRESI SIGNIFIKAN'
ELSE
NOTE 'GAGAL TOLAK H0, MAKA KOEFISIEN PARAMETER MODEL REGRESI SIGNIFIKAN'
ENDIF

NOTE _________________________________________________________________________
#UJI PARSIAL A
NOTE UJI PARSIAL INTERSEP
LET ALPHA2=1-(ALPHA/2)

let ai=0
DO I=1:N
LET ai=ai+((X(i)-XBAR)^2)
enddo

LET SE_A=sqrt(MSE/ai)
LET THIT_A=A/SE_A
INVCDF ALPHA2 TTAB;
	T DFE.
PRINT SE_A THIT_A TTAB

IF ABS(THIT_A) > TTAB
NOTE 'TOLAK H0, MAKA KOEFISIEN PARAMETER A BERPENGARUH SIGNIFIKAN TERHADAP MODEL'
ELSE
NOTE 'GAGAL TOLAK H0, MAKA KOEFISIEN PARAMETER A TIDAK BERPENGARUH SIGNIFIKAN TERHADAP MODEL'
ENDIF

#menghitung p-value A
let THIT_A=abs(THIT_A)
CDF THIT_A PV_A;
	T DFE.
let pvalue_A=2*(1-PV_A)

print pvalue_A alpha
if (pvalue_A < alpha)
NOTE 'TOLAK H0, MAKA KOEFISIEN PARAMETER A BERPENGARUH SIGNIFIKAN TERHADAP MODEL'
ELSE
NOTE 'GAGAL TOLAK H0, MAKA KOEFISIEN PARAMETER A TIDAK BERPENGARUH SIGNIFIKAN TERHADAP MODEL'
ENDIF

NOTE _________________________________________________________________________

#UJI PARSIAL B
NOTE UJI PARSIAL SLOPE

let bi=0
DO I=1:N
LET bi=bi+((X(I)-XBAR)^2)
enddo
LET SE_B=sqrt(MSE)/sqrt(bi)
LET THIT_B=B/SE_B
INVCDF ALPHA2 TTAB;
	T DFE.
PRINT SE_B THIT_B TTAB

IF ABS(THIT_B) > TTAB
NOTE 'TOLAK H0, MAKA KOEFISIEN PARAMETER B BERPENGARUH SIGNIFIKAN TERHADAP MODEL'
ELSE
NOTE 'GAGAL TOLAK H0, MAKA KOEFISIEN PARAMETER B TIDAK BERPENGARUH SIGNIFIKAN TERHADAP MODEL'
ENDIF

#menghitung p-value B
let THIT_B=abs(THIT_B)
CDF THIT_B PV_B;
	T DFE.
let pvalue_B=2*(1-PV_B)

print pvalue_B alpha
if (pvalue_B < alpha)
NOTE 'TOLAK H0, MAKA KOEFISIEN PARAMETER B BERPENGARUH SIGNIFIKAN TERHADAP MODEL'
ELSE
NOTE 'GAGAL TOLAK H0, MAKA KOEFISIEN PARAMETER B TIDAK BERPENGARUH SIGNIFIKAN TERHADAP MODEL'
ENDIF
